from datetime import datetime

from app.schemas.product import ProductCreate, ProductUpdate


def list_products(db: Session, search: str | None = None) -> list[Product]:
    query = db.query(Product).order_by(Product.id.desc())

    if search:
        pattern = f"%{search.strip()}%"
        query = query.filter(
            Product.name.ilike(pattern)
            | Product.product_code.ilike(pattern)
            | Product.sale_price.ilike(pattern)
        )

    return query.all()


def get_product(db: Session, product_id: int) -> Product | None:
    return db.query(Product).filter(Product.id == product_id).first()


def get_product_by_code(db: Session, product_code: str) -> Product | None:
    return db.query(Product).filter(Product.product_code == product_code).first()


def create_product(db: Session, product_data: ProductCreate) -> Product:
    product = Product(
        **product_data.model_dump(),
        updated_at=datetime.utcnow(),
    )
    db.add(product)
    db.commit()
    db.refresh(product)
    return product


def update_product(db: Session, product: Product, product_data: ProductUpdate) -> Product:
    update_data = product_data.model_dump(exclude_unset=True)

    for field, value in update_data.items():
        setattr(product, field, value)

    product.updated_at = datetime.utcnow()

    db.commit()
    db.refresh(product)
    return product


def delete_product(db: Session, product: Product) -> None:

        code = f"ORD-{next_number:04d}"

def _add_months(source: date, months: int) -> date:

def _period_bounds(period: str, anchor: date, offset: int = 0) -> tuple[date, date]:
    if period == "yearly":
        year = anchor.year + offset
        start = date(year, 1, 1)
        end = anchor if offset == 0 else date(year, 12, 31)
        return start, end

    if period == "monthly":
        month_anchor = _add_months(anchor, offset)
        start = date(month_anchor.year, month_anchor.month, 1)
        next_month = _add_months(start, 1)
        end = next_month - timedelta(days=1)

        if offset == 0:
            end = anchor

        return start, end

    if offset == 0:
        end = anchor
        start = anchor - timedelta(days=6)
        return start, end

    previous_end = anchor - timedelta(days=7)
    previous_start = previous_end - timedelta(days=6)
    return previous_start, previous_end


def _in_range(item_date: date | None, start: date, end: date) -> bool:
    if item_date is None:
        return False

    return start <= item_date <= end


def _growth(current: float, previous: float) -> float:
    if previous == 0:
        return 0.0 if current == 0 else 100.0

    return round(((current - previous) / previous) * 100, 1)


def _period_totals(
    orders: list[dict[str, Any]],
    order_item_totals: dict[str, float],
    period: str,
    anchor: date,
    offset: int = 0,
) -> dict[str, float]:
    start, end = _period_bounds(period, anchor, offset)

def _orders_by_period(
    orders: list[dict[str, Any]],
    order_item_totals: dict[str, float],
    period: str,
    anchor: date,
) -> list[dict[str, Any]]:
    if period == "yearly":
        rows = []

        for month in range(1, anchor.month + 1):
            current_count = 0
            current_amount = 0.0

            for order in orders:
                order_date = _order_date(order)

                if order_date and order_date.year == anchor.year and order_date.month == month:
                    current_count += 1
                    current_amount += _order_total(order, order_item_totals)

            rows.append(
                {
                    "date": f"{anchor.year}-{month:02d}",
                    "label": f"{month:02d}",
                    "count": current_count,
                    "amount": current_amount,
                }
            )

        return rows

    start_date, end_date = _period_bounds(period, anchor, 0)
    days = (end_date - start_date).days + 1
    rows = []

    for offset in range(days):
        current_date = start_date + timedelta(days=offset)
        current_count = 0
        current_amount = 0.0

        for order in orders:
            order_date = _order_date(order)

            if order_date == current_date:
                current_count += 1
                current_amount += _order_total(order, order_item_totals)

        rows.append(
            {
                "date": current_date.isoformat(),
                "label": current_date.strftime("%m/%d"),
                "count": current_count,
                "amount": current_amount,
            }
        )

    return rows

def normalize_username(username: str) -> str:
    return (username or "").strip().lower()


def get_user_by_username(db: Session, username: str) -> User | None:
    normalized_username = normalize_username(username)
    if not normalized_username:
        return None

    return db.query(User).filter(User.username == normalized_username).first()


def get_user_by_id(db: Session, user_id: int) -> User | None:
    return db.query(User).filter(User.id == user_id).first()


def create_user(db: Session, user_data: UserCreate) -> User:
    user = User(
        username=normalize_username(user_data.username),
        full_name=user_data.full_name,
        role=user_data.role,
        hashed_password=hash_password(user_data.password),
        is_active=True,
    )

    db.add(user)
    db.commit()
    db.refresh(user)

    return user


def is_password_valid(password: str, hashed_password: str) -> bool:
    if not password or not hashed_password:
        return False

    try:
        return verify_password(password, hashed_password)
    except Exception:
        return False


def authenticate_user(db: Session, username: str, password: str) -> User | None:
    user = get_user_by_username(db, username)

    if not user:
        return None

    if not is_password_valid(password=password, hashed_password=user.hashed_password):
        return None

    return user

from sqlalchemy import inspect, text
from app.database import engine


def _sqlite_add_column(table_name: str, column_name: str, column_definition: str) -> None:
    inspector = inspect(engine)
    existing_columns = {column["name"] for column in inspector.get_columns(table_name)}
    if column_name in existing_columns:
        return

    with engine.begin() as connection:
        connection.execute(text(f"ALTER TABLE {table_name} ADD COLUMN {column_name} {column_definition}"))


def ensure_database_schema() -> None:
    """Small SQLite-safe schema upgrades used during the web migration.

    This avoids losing local data while the old desktop schema is being adapted.
    Later we can replace this with Alembic migrations, but for the current fast
    migration path this keeps the database compatible without manual edits.
    """
    if not str(engine.url).startswith("sqlite"):
        return

    inspector = inspect(engine)
    table_names = set(inspector.get_table_names())

    if "users" in table_names:
        _sqlite_add_column("users", "email", "VARCHAR")
        _sqlite_add_column("users", "updated_at", "DATETIME")

    if "customers" in table_names:
        _sqlite_add_column("customers", "updated_at", "DATETIME")

    if "products" in table_names:
        _sqlite_add_column("products", "updated_at", "DATETIME")
